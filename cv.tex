\newcommand{\rsection}[1]{%
  \vspace{4pt}%
  {\color{accent}\large\bfseries #1}%
  \par\vspace{-4pt}%
  {\color{accent}\rule{\linewidth}{1pt}}%
  \vspace{2pt}%
}

  \vspace{18pt}%
  {\color{white}\bfseries\uppercase{#1}}%

  \vspace{4pt}%

%% ── Entrée diplôme (inline compact) ───────────────────────────────────────
\newcommand{\cvdiploma}[3]{%
  % #1 = année   #2 = diplôme (gras)   #3 = école
  \vspace{2pt}
  {\small\color{gray}#1}\qquad{\small\bfseries #2}%
  \def\school{#3}\ifx\school\empty\else{ \small\itshape — #3}\fi\par
  \vspace{1pt}%
}

%% ── Entrée expérience / diplôme ────────────────────────────────────────────
\newcommand{\cventry}[4]{%
  % #1 = période   #2 = titre   #3 = lieu/entreprise   #4 = description
  \vspace{3pt}\noindent
  \makebox[1.8cm][l]{\small\color{gray}#1}\hspace{6pt}%
  \begin{minipage}[t]{\dimexpr\linewidth-2.2cm}
    {\bfseries #2}, \small\itshape #3\\[2pt]
    \small\upshape #4
  \end{minipage}
  \vspace{3pt}%
}

%% ── Catégorie de compétences ───────────────────────────────────────────────

  % #1 = catégorie   #2 = items séparés par \\ dans une itemize

  \begin{itemize}[leftmargin=10pt, itemsep=0pt, topsep=0pt, parsep=0pt, label={\color{accent}\tiny\textbullet}]
    \color{textside}\small #2
  \end{itemize}
}
